class Solution {
    public int peakIndexInMountainArray(int[] nums) {
       int start=0;
       int end=nums.length-1;
       while(start<end){
            int mid=start+(end-start)/2;
            if(nums[mid]<nums[mid+1]){
                start=mid+1;
            }
            else{
                end=mid;
            }
        }
        return start;
        //return end; ////you can return the end also
    } 
}
